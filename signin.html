require("dotenv").config(); const express = require("express"); const router = express.Router(); const mongoose = require("mongoose"); const bcrypt = require("bcryptjs"); // Asire ou gen yon modèl itilizatè (User model) const User = require("./models/User"); // Chanje chemen selon pwojè ou // Route POST pou signin router.post("/signin", async (req, res) => { try { const { username, password } = req.body; // Verifye si username egziste const user = await User.findOne({ username: username }); if (!user) { return res.status(400).json({ message: "Non itilizatè pa egziste" }); } // Verifye modpas la avèk bcrypt const isMatch = await bcrypt.compare(password, user.password); if (!isMatch) { return res.status(400).json({ message: "Modpas pa kòrèk" }); } // Si tout bagay bon, voye repons siksè (ou ka ajoute JWT token tou) res.status(200).json({ message: "Otantifikasyon reyisi", user: user.username }); } catch (err) { console.error("Erè signin:", err); res.status(500).json({ message: "Erè sèvè entè" }); } }); module.exports = router;
